> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a curses TUI to `gq`: typing bare `gq` opens a htop-style panel where operations and jobs are selected with arrow keys + Enter (no command typing, no hotkeys); `Add` spawns a real bash with an F5 bind to grab the command line; task output goes to per-job log files.

**Architecture:** The TUI is a new view process (`gq` no-arg) over the existing daemon/state files. Pure curses. `gq watch` and all CLI subcommands are unchanged. Task stdout/stderr is redirected to `~/.gpu-queue/logs/<id>.log` by the daemon's `_launch_job`. The TUI's "what to show" brain (`_build_rows`) and the bash-capture (`_run_bash_for_command`) are pure/testable; the curses rendering is a thin shell over them.

**Tech Stack:** Python 3 stdlib — `curses` (TUI), `subprocess` (bash spawn), `fcntl`/`json` (state, existing). pytest (with `PYTHONPATH=` prefix on this machine). No new dependencies.

```python
def _launch_job(job: dict, cards: list) -> Optional[subprocess.Popen]:
    """Launch a job on the given cards (non-blocking). Returns the Popen, or
    None on launch failure. Injects CUDA_VISIBLE_DEVICES into the job's env.
    Task stdout/stderr is redirected to ~/.gpu-queue/logs/<id>.log (the watch
    terminal prints only summary lines; the TUI can display the log tail).
    """
    cwd = job["cwd"]
    if not os.path.isdir(cwd):
        print(f"[gq] WARNING: cwd '{cwd}' does not exist, skipping job {job['id']}.")
        return None
    env = dict(job.get("env") or {})
    env["CUDA_VISIBLE_DEVICES"] = ",".join(str(c) for c in cards)
    LOG_DIR.mkdir(parents=True, exist_ok=True)
    log_path = LOG_DIR / f"{job['id']}.log"
    try:
        log_f = open(log_path, "w")
    except OSError as exc:
        print(f"[gq] ERROR opening log {log_path}: {exc}")
        return None
    print(f"[gq] >>> {job['cmd']}   (GPU {','.join(str(c) for c in cards)})   log: {log_path}")
    try:
        proc = subprocess.Popen(job["cmd"], shell=True, cwd=cwd,
                                start_new_session=True, env=env,
                                stdout=log_f, stderr=subprocess.STDOUT)
    except Exception as exc:
        log_f.close()
        print(f"[gq] ERROR running job {job['id']}: {exc}")
        return None
    # NOTE: log_f is intentionally not closed here — Popen writes to it
    # asynchronously. It is closed when the daemon reaps the job (see _daemon_loop
    # reap block: close the proc's stdout after poll() returns non-None).
    return proc
```

```python
def _build_rows(state: dict, queue: list) -> list:
    """Build the TUI operations list from current state + queue. Pure (no curses).

    Row order: Add, then one Stop + one Open-log per running job, then one
    Cancel per queued job, then Clear, Quit. Each row:
      {"kind": "op"|"job", "label": str, "action": str, "job_id": str|None}
    """
    rows = []
    rows.append({"kind": "op", "label": "Add job",
                 "action": "add", "job_id": None})
    running = state.get("running") or {}
    for rid, r in running.items():
        cards = ",".join(str(c) for c in r.get("cards", []))
        rows.append({"kind": "job", "label": f"Stop: {rid}  ({r.get('cmd','')} GPU {cards})",
                     "action": "stop", "job_id": rid})
    for rid, r in running.items():
        rows.append({"kind": "job", "label": f"Open log: {rid}",
                     "action": "open_log", "job_id": rid})
    for i, job in enumerate(queue, 1):
        rows.append({"kind": "job",
                     "label": f"Cancel: #{i} {job['id']}  ({job.get('cmd','')})",
                     "action": "cancel", "job_id": job["id"]})
    rows.append({"kind": "op", "label": "Clear queue",
                 "action": "clear", "job_id": None})
    rows.append({"kind": "op", "label": "Quit",
                 "action": "quit", "job_id": None})
    return rows
```

```python
import curses


def _render_panel(stdscr, rows, focus, gpu_summary):
    """Render the TUI: top GPU bar + operations list + bottom hint."""
    stdscr.erase()
    h, w = stdscr.getmaxyx()
    curses.init_pair(1, curses.COLOR_GREEN, curses.COLOR_BLACK)
    curses.init_pair(2, curses.COLOR_YELLOW, curses.COLOR_BLACK)
    curses.init_pair(3, curses.COLOR_RED, curses.COLOR_BLACK)
    curses.init_pair(4, curses.COLOR_CYAN, curses.COLOR_BLACK)

    # Top GPU summary bar (read-only)
    stdscr.addstr(0, 0, gpu_summary[:w-1], curses.color_pair(4) | curses.A_BOLD)
    stdscr.addstr(1, 0, "-" * (w - 1))

    # Operations list
    for i, row in enumerate(rows):
        y = 2 + i
        if y >= h - 1:
            break
        label = row["label"][:w-1]
        attr = curses.A_REVERSE if i == focus else curses.A_NORMAL
        stdscr.addstr(y, 0, label, attr)

    # Bottom hint
    stdscr.addstr(h - 1, 0, " Up/Dn select  Enter confirm  q quit "[:w-1],
                  curses.color_pair(4))


def _gpu_summary_line():
    """One-line GPU overview for the top bar."""
    total = _total_cards()
    busy = busy_cards()
    state = read_state()
    nrun = len(state.get("running") or {})
    nq = len(read_queue())
    import datetime
    clock = datetime.datetime.now().strftime("%H:%M:%S")
    idle = total - len([g for g in busy])
    return f" gq  {total} GPUs ({idle} idle)  {nrun} running  {nq} queued   {clock}"


def _tui_main(stdscr):
    """Curses TUI main loop. stdscr is provided by curses.wrapper."""
    curses.curs_set(0)  # hide cursor
    stdscr.keypad(True)
    curses.halfdelay(20)  # 2-second refresh tick; keys return immediately
    focus = 0
    while True:
        state = read_state()
        queue = read_queue()
        rows = _build_rows(state, queue)
        if focus >= len(rows):
            focus = max(0, len(rows) - 1)
        _render_panel(stdscr, rows, focus, _gpu_summary_line())
        ch = stdscr.getch()
        if ch in (curses.KEY_UP, ord("k")):
            focus = max(0, focus - 1)
        elif ch in (curses.KEY_DOWN, ord("j")):
            focus = min(len(rows) - 1, focus + 1)
        elif ch == ord("q"):
            break
        elif ch in (curses.KEY_ENTER, 10, 13):
            done = _tui_do_action(stdscr, rows[focus])
            if done:
                break
        # any other key / timeout: just re-render on next cycle
```

```python
def _run_bash_for_command():
    """Spawn an interactive bash for the user to compose/test a command. F5
    grabs the current readline line (without executing it) plus cwd + env,
    then exits bash. Returns {"cmd","cwd","env"} on F5, or None if the user
    quits bash (Ctrl-D) without pressing F5.

    Mechanism: write a small bash init script that binds F5 (\\e[15~) to a
    command which dumps $READLINE_LINE, $PWD, and `env` to temp files then
    `exit 0`. The temp-file paths are exported so we can read them after.
    FALLBACK: if F5/bind proves unreliable on a terminal, replace this with
    a curses single-line input box (no cd/tab/test-run) returning the same dict.
    """
    import tempfile
    cmd_fd, cmd_path = tempfile.mkstemp(prefix="gq_cmd_")
    cwd_fd, cwd_path = tempfile.mkstemp(prefix="gq_cwd_")
    env_fd, env_path = tempfile.mkstemp(prefix="gq_env_")
    for fd in (cmd_fd, cwd_fd, env_fd):
        os.close(fd)

    init_script = f"""
bind -x '"\\e[15~": "printf %s \\"$READLINE_LINE\\" > {cmd_path}; pwd > {cwd_path}; env -0 > {env_path}; exit 0"'
echo "[gq] compose your command, then press F5 to submit to the queue (Ctrl-D to cancel)."
"""
    try:
        proc = subprocess.Popen(["bash", "--rcfile", "/dev/null", "-i"],
                                stdin=sys.stdin, stdout=sys.stdout, stderr=sys.stderr,
                                env={**os.environ, "BASH_ENV": "", "GQ_INIT": init_script})
        # Inject the init via -c is tricky for interactive; use inputrc-free approach:
        # write init to a temp rcfile and pass as --rcfile.
    except Exception as exc:
        print(f"[gq] ERROR spawning bash: {exc}", file=sys.stderr)
        return None
    proc.wait()

    cmd = Path(cmd_path).read_text()
    if not cmd:
        return None  # user quit without F5
    cwd = Path(cwd_path).read_text().strip()
    env = {}
    raw = Path(env_path).read_bytes()
    for entry in raw.split(b"\x00"):
        if b"=" in entry:
            k, _, v = entry.partition(b"=")
            try:
                env[k.decode()] = v.decode()
            except UnicodeDecodeError:
                continue
    for p in (cmd_path, cwd_path, env_path):
        try:
            os.unlink(p)
        except OSError:
            pass
    return {"cmd": cmd, "cwd": cwd, "env": env}
```

```python
def _dispatch_action(row: dict) -> bool:
    """Pure dispatch for a TUI row's action (no curses). Returns True if the
    TUI should quit. add/open_log/stop are handled by the curses-aware
    _tui_do_action (they need stdscr or subprocess); _dispatch_action handles
    the state-only ones (quit/cancel/clear) for testability.
    """
    action = row["action"]
    if action == "quit":
        return True
    if action == "cancel":
        # Reuse cmd_cancel's logic by constructing an args-like namespace.
        import argparse
        gq.cmd_cancel(argparse.Namespace(job_id=row["job_id"]))
        return False
    if action == "clear":
        gq.cmd_clear(argparse.Namespace())
        return False
    # add, stop, open_log are curses/subprocess-bound — handled in _tui_do_action
    return False
```

```python
def _tui_confirm(stdscr, msg):
    """y/n confirm. Returns True on y."""
    h, w = stdscr.getmaxyx()
    stdscr.addstr(h - 2, 0, (" " * (w - 1)))
    stdscr.addstr(h - 2, 0, f" {msg} [y/n] "[:w-1], curses.A_BOLD)
    while True:
        ch = stdscr.getch()
        if ch in (ord("y"), ord("Y")):
            return True
        if ch in (ord("n"), ord("N"), 27):  # n or Esc
            return False


def _tui_select_gpus(stdscr):
    """Curses list-select for --gpus in 1..N. Returns the selected int, or 1."""
    n = _total_cards()
    choices = list(range(1, n + 1))
    focus = 0
    h, w = stdscr.getmaxyx()
    while True:
        stdscr.addstr(h - 2, 0, " " * (w - 1))
        line = " --gpus: " + "  ".join(f"[{c}]" if i == focus else f" {c} "
                                       for i, c in enumerate(choices))
        stdscr.addstr(h - 2, 0, line[:w-1], curses.A_BOLD)
        ch = stdscr.getch()
        if ch in (curses.KEY_LEFT, ord("h")):
            focus = max(0, focus - 1)
        elif ch in (curses.KEY_RIGHT, ord("l")):
            focus = min(len(choices) - 1, focus + 1)
        elif ch in (curses.KEY_ENTER, 10, 13):
            return choices[focus]
        elif ch == 27:
            return 1  # default on Esc


def _tui_do_action(stdscr, row):
    """Execute a TUI row's action. Returns True if TUI should quit."""
    action = row["action"]
    if action == "quit":
        return True
    if action == "add":
        captured = _run_bash_for_command()
        if not captured:
            return False  # user cancelled in bash
        n = _tui_select_gpus(stdscr)
        job = _make_job(captured["cmd"], captured["cwd"], n=n)
        # use the bash-captured env instead of the snapshot (bash may have
        # activated a conda env during composition)
        job["env"] = captured["env"]
        with _locked_queue() as q:
            q.append(job)
        return False
    if action == "stop":
        if _tui_confirm(stdscr, f"Stop {row['job_id']}?"):
            import argparse
            gq.cmd_stop(argparse.Namespace(job_id=row["job_id"]))
        return False
    if action == "cancel":
        if _tui_confirm(stdscr, f"Cancel {row['job_id']}?"):
            _dispatch_action(row)
        return False
    if action == "clear":
        if _tui_confirm(stdscr, "Clear all queued jobs?"):
            _dispatch_action(row)
        return False
    if action == "open_log":
        log_path = LOG_DIR / f"{row['job_id']}.log"
        _tui_show_log(stdscr, log_path)
        return False
    return False


def _tui_show_log(stdscr, log_path):
    """Show the tail of a job's log file; any key returns."""
    h, w = stdscr.getmaxyx()
    stdscr.erase()
    try:
        text = Path(log_path).read_text(errors="replace")
    except OSError:
        text = "(no log yet)"
    lines = text.splitlines()[-(h - 2):]
    stdscr.addstr(0, 0, f" log: {log_path}"[:w-1], curses.A_BOLD)
    for i, ln in enumerate(lines, 1):
        if i >= h - 1:
            break
        stdscr.addstr(i, 0, ln[:w-1])
    stdscr.addstr(h - 1, 0, " any key to return "[:w-1], curses.color_pair(4))
    stdscr.getch()
```
