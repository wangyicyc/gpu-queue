**Goal:** Add a `gq stop` subcommand that stops the currently-running job from any window (SIGKILL its process group) while the daemon stays alive and moves on to the next queued job; plus fix README's tmux framing (optional, not required).

**Architecture:** `cmd_stop` reads `state.json`'s `running.pid` and calls `os.killpg(os.getpgid(pid), signal.SIGKILL)` — the exact same operation the existing crash-recovery block uses (gq:417-443).     gq.cmd_stop(_args())

```python
def test_cmd_stop_kills_running_job(monkeypatch, capsys):
    """gq stop SIGKILLs the running job's process group via its pid."""
    gq.write_state({"daemon_pid": None,
                    "running": {"id": "ab12", "cmd": "x", "pid": 12345}})
    calls = {"getpgid": None, "killpg": None}

    def fake_getpgid(pid):
        calls["getpgid"] = pid
        return 99999  # the process group id

    def fake_killpg(pgid, sig):
        calls["killpg"] = (pgid, sig)

    monkeypatch.setattr(gq.os, "getpgid", fake_getpgid)
    monkeypatch.setattr(gq.os, "killpg", fake_killpg)
    gq.cmd_stop(_args())
    out = capsys.readouterr().out
    assert calls["getpgid"] == 12345
    assert calls["killpg"] == (99999, signal.SIGKILL)
    assert "stopped job ab12" in out
    assert "12345" in out


def test_cmd_stop_pid_already_dead(monkeypatch, capsys):
    """If the pid is already gone (ProcessLookupError), report gracefully."""
    gq.write_state({"daemon_pid": None,
                    "running": {"id": "ab12", "cmd": "x", "pid": 12345}})

    def fake_getpgid(pid):
        raise ProcessLookupError

    killed = []
    monkeypatch.setattr(gq.os, "getpgid", fake_getpgid)
    monkeypatch.setattr(gq.os, "killpg", lambda pgid, sig: killed.append((pgid, sig)))
    gq.cmd_stop(_args())
    out = capsys.readouterr().out
    assert "already finished" in out or "not found" in out
    assert killed == []  # must NOT have called killpg


def test_cmd_stop_pid_none_treated_as_no_job(capsys):
    """running entry exists but pid is None (job not fully started) → no job."""
    gq.write_state({"daemon_pid": None,
                    "running": {"id": "ab12", "cmd": "x", "pid": None}})
    gq.cmd_stop(_args())
    out = capsys.readouterr().out
```

```python
def cmd_stop(args) -> None:
    """Stop the currently-running job (SIGKILL its process group).

    Reads state.running.pid and kills the whole process group — the same
    operation crash-recovery uses on daemon restart. The daemon is NOT
    signaled: it is blocked in proc.wait(), which returns when the job is
    killed, then it clears running and continues to the next queued job.
    Run from any window; no need to switch to the daemon terminal.
    """
    state = read_state()
    running = state.get("running")
    rpid = running.get("pid") if running else None

    if not rpid:
        print("[gq] no job currently running.")
        return

    try:
        pgid = os.getpgid(rpid)
        os.killpg(pgid, signal.SIGKILL)
    except ProcessLookupError:
        print(f"[gq] job {running.get('id')} already finished "
              f"(pid {rpid} not found).")
        return
    except PermissionError as exc:
        print(f"[gq] ERROR: cannot stop job {running.get('id')} "
              f"(pid {rpid}): {exc}")
        return
    print(f"[gq] stopped job {running.get('id')} (pid {rpid}).")
```
